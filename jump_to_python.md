```python

# 중복을 허용하지 않는다.
# 순서가 없다(unodered)
 # 중복을 허용하지 않는 set의 특징은 자료형의 중복을 제거하기 위한 필터 역할로 종종 사용
 # 교집합, 합집합, 차집합을 구할 때 유용

print(s1-s2) #차집합 = s1.difference(s2)
```
